```dart

    // 构造 0xAA ... 0x55 格式数据包
}

**解决方案**:

**方案A（推荐）**: 修改软件端，添加文本协议支持
```dart
// 新增方法
Future<bool> sendTextCommand(String command) async {
  if (!isConnected || _writeCharacteristic == null) {
    return false;
  }

  try {
    String cmd = command.endsWith('\n') ? command : '$command\n';
    List<int> bytes = utf8.encode(cmd);

    await _writeCharacteristic!.write(bytes, withoutResponse: true);
    print('[JDY08] Sent: $cmd');
    return true;
  } catch (e) {
    print('[JDY08] Send failed: $e');
    return false;
  }
}

// 使用示例
await jdy08.sendTextCommand('FAN:50');
await jdy08.sendTextCommand('LED:1:255:0:0');

**方案B**: 修改硬件端支持二进制协议（需要重写protocol.c）

### 修改JDY08BluetoothService

在`lib/services/jdy08_bluetooth_service.dart`中添加以下方法：

```dart
import 'dart:convert';  // 添加导入

/// 发送文本命令（适配硬件文本协议）
/// 
/// [command] 命令字符串，如 "FAN:50"
/// 返回是否发送成功
Future<bool> sendTextCommand(String command) async {
  if (!isConnected || _writeCharacteristic == null) {
    print('[JDY08] Not connected');
    return false;
  }

  try {
    // 确保命令以换行符结尾
    String cmd = command.endsWith('\n') ? command : '$command\n';
    
    // 转换为字节数组
    List<int> bytes = utf8.encode(cmd);
    
    // 发送数据
    await _writeCharacteristic!.write(bytes, withoutResponse: true);
    
    print('[JDY08] Sent: ${cmd.trim()}');
    return true;
  } catch (e) {
    print('[JDY08] Send failed: $e');
    return false;
  }
}

/// 便捷方法：设置风扇速度（文本协议）
Future<bool> setFanSpeedText(int percent) async {
  if (percent < 0) percent = 0;
  if (percent > 100) percent = 100;
  return await sendTextCommand('FAN:$percent');
}

/// 便捷方法：设置LED颜色（文本协议）
Future<bool> setLedColorText(int strip, int r, int g, int b) async {
  if (strip < 1 || strip > 4) return false;
  if (r < 0) r = 0; if (r > 255) r = 255;
  if (g < 0) g = 0; if (g > 255) g = 255;
  if (b < 0) b = 0; if (b > 255) b = 255;
  
  return await sendTextCommand('LED:$strip:$r:$g:$b');
}

/// 便捷方法：播放音频（文本协议）
Future<bool> playAudioText(int index) async {
  if (index < 0 || index >= 15) return false;
  return await sendTextCommand('AUDIO:PLAY:$index');
}

/// 便捷方法：查询风扇状态
Future<bool> queryFanStatus() async {
  return await sendTextCommand('GET:FAN');
}

/// 便捷方法：查询音频状态
Future<bool> queryAudioStatus() async {
  return await sendTextCommand('GET:AUDIO');
}

/// 解析接收到的文本响应
void _parseTextResponse(String response) {
  response = response.trim();
  
  // 解析风扇状态: FAN:50
  if (response.startsWith('FAN:')) {
    try {
      int speed = int.parse(response.substring(4));
      _statusController.add({
        'type': 'fan',
        'speed': speed,
        'timestamp': DateTime.now().millisecondsSinceEpoch,
      });
    } catch (e) {
      print('[JDY08] Parse FAN response error: $e');
    }
  }
  
  // 解析音频状态: AUDIO:PLAYING:80:0:15
  else if (response.startsWith('AUDIO:')) {
    try {
      List<String> parts = response.substring(6).split(':');
      if (parts.length >= 4) {
        _statusController.add({
          'type': 'audio',
          'state': parts[0],       // PLAYING/PAUSED/STOPPED
          'volume': int.parse(parts[1]),
          'currentFile': int.parse(parts[2]),
          'totalFiles': int.parse(parts[3]),
          'timestamp': DateTime.now().millisecondsSinceEpoch,
        });
      }
    } catch (e) {
      print('[JDY08] Parse AUDIO response error: $e');
    }
  }
  
  // 解析操作确认: OK:FAN:50
  else if (response.startsWith('OK:')) {
    _statusController.add({
      'type': 'ok',
      'message': response.substring(3),
      'timestamp': DateTime.now().millisecondsSinceEpoch,
    });
  }
  
  // 解析错误信息: ERR:message
  else if (response.startsWith('ERR:')) {
    _statusController.add({
      'type': 'error',
      'message': response.substring(4),
      'timestamp': DateTime.now().millisecondsSinceEpoch,
    });
  }
}

// 在 connectToDevice 方法中修改数据接收处理
// 第151行左右，替换为：
_notificationSubscription = _notifyCharacteristic!.lastValueStream.listen((value) {
  if (value.isNotEmpty) {

    // 尝试解析为文本

      _parseTextResponse(text);
    } catch (e) {
      // 如果不是UTF-8文本，可能是二进制数据

class DeviceControlWidget extends StatelessWidget {
  final JDY08BluetoothService _jdy08 = JDY08BluetoothService();

          },
        ),

              color.red,
              color.green,
              color.blue,
            );
          },
        ),
        
        // 音频控制按钮
        Row(
          children: [
            IconButton(
              icon: Icon(Icons.play_arrow),
              onPressed: () => _jdy08.playAudioText(0),
            ),
            IconButton(
              icon: Icon(Icons.stop),
              onPressed: () => _jdy08.sendTextCommand('AUDIO:STOP'),

              onPressed: () => _jdy08.sendTextCommand('AUDIO:NEXT'),
            ),
          ],
        ),
        
        // 状态监听
        StreamBuilder<Map<String, dynamic>>(
          stream: _jdy08.statusStream,
          builder: (context, snapshot) {
            if (!snapshot.hasData) return SizedBox.shrink();

            var status = snapshot.data!;
            if (status['type'] == 'fan') {
              return Text('风扇速度: ${status['speed']}%');
            } else if (status['type'] == 'audio') {
              return Text('音频: ${status['state']} - ${status['currentFile']}/${status['totalFiles']}');
            } else if (status['type'] == 'error') {
              return Text('错误: ${status['message']}', style: TextStyle(color: Colors.red));
            }

            return SizedBox.shrink();
          },
        ),
